About ACOE
The Alameda County Office of Education, under Superintendent Alysse Castro, operates schools serving the county's most vulnerable students — including Court Schools at the JJC, Quest Academies, Community Schools, and the Parenting Teens Program.
www.acoe.org ↗
About AC Probation
The Alameda County Probation Department, under Chief Brian K. Ford, provides supervision, intervention, and support for justice-involved youth through its Positive Youth Development division — including YVRC, the Ubuntu Justice Initiative, and the My FUTURE Center.
probation.acgov.org ↗
Our Vision
"We equip the most vulnerable students and those that serve them with the tools to thrive, erasing the predetermination of failure for children from communities of violence and poverty."
Sources: AC Probation (Oakland Youth & County JJCPA), ACOE Schools, City of Oakland DVP

const SHEET_ID = 'YOUR_SHEET_ID'; const SHEET_NAME = 'Resources'; const COLS = ['id','title','cat','catL','icon','th','ages','ageMin','ageMax', 'genders','riskFactors','populations','urgency','zipcodes', 'desc','tags','contact','addr','avail','link','src','srcL']; function doGet(e) { const p = e.parameter, cb = p.callback || 'cb'; const ss = SpreadsheetApp.openById(SHEET_ID); try { if (p.action === 'ping') return jp(cb, {ok:true}); if (p.action === 'read') { const sh = ss.getSheetByName(SHEET_NAME); if (!sh) return jp(cb, []); const v = sh.getDataRange().getValues(); if (v.length < 2) return jp(cb, []); const hdrs = v[0]; return jp(cb, v.slice(1).filter(r=>r[0]).map(row => { const obj = {}; hdrs.forEach((h,i) => { let val = String(row[i]===undefined?'':row[i]); if (['genders','riskFactors','populations','tags'].includes(h)) obj[h] = val ? val.split('|').map(s=>s.trim()).filter(Boolean) : []; else if (['ageMin','ageMax','id'].includes(h)) obj[h] = (val===''||val==='null') ? null : Number(val); else obj[h] = val; }); return obj; })); } if (p.action === 'write') { const d = JSON.parse(decodeURIComponent(p.data)); const sh = getOrCreate(ss); const v = sh.getDataRange().getValues(); const row = makeRow(d); let found = false; for (let i=1;i<v.length;i++) { if (String(v[i][0])===String(d.id)) { sh.getRange(i+1,1,1,COLS.length).setValues([row]); found=true; break; } } if (!found) sh.appendRow(row); return jp(cb, {ok:true}); } if (p.action === 'delete') { const sh = ss.getSheetByName(SHEET_NAME); if (sh) { const v = sh.getDataRange().getValues(); for (let i=1;i<v.length;i++) { if (String(v[i][0])===String(p.id)) { sh.deleteRow(i+1); break; } } } return jp(cb, {ok:true}); } if (p.action === 'seed') { const records = JSON.parse(decodeURIComponent(p.data)); const sh = getOrCreate(ss); const lastRow = sh.getLastRow(); if (lastRow > 1) sh.deleteRows(2, lastRow-1); records.forEach(r => sh.appendRow(makeRow(r))); return jp(cb, {ok:true, count:records.length}); } } catch(err) { return jp(cb, {ok:false, error:err.message}); } return jp(cb, {ok:false, error:'unknown action'}); } function makeRow(d) { return COLS.map(c => { const v = d[c]; if (Array.isArray(v)) return v.join('|'); if (v===null||v===undefined) return ''; return String(v); }); } function getOrCreate(ss) { let sh = ss.getSheetByName(SHEET_NAME); if (!sh) { sh = ss.insertSheet(SHEET_NAME); sh.appendRow(COLS); sh.getRange(1,1,1,COLS.length) .setFontWeight('bold').setBackground('#0D9E75').setFontColor('#ffffff'); sh.setFrozenRows(1); sh.setColumnWidth(2,260); sh.setColumnWidth(15,300); sh.setColumnWidth(17,200); } return sh; } function jp(cb, data) { return ContentService .createTextOutput(cb+'('+JSON.stringify(data)+');') .setMimeType(ContentService.MimeType.JAVASCRIPT); }
